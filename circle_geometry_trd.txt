1. TECHNICAL OVERVIEW

This document specifies the architecture, component design, state
management, data models, simulation logic, gamification implementation,
audio pipeline integration, and quality standards for the "Circle"
interactive lesson module (Lesson 6.1) within Intellia SG's Grade 1 Math
program.

The module is a React 18 application (Vite + JSX), structured identically
to the reference repository https://github.com/dsamyak/wordproblem_sub,
and styled to match https://wordproblem-sub.vercel.app/. It will be
embedded in WordPress at:

  https://intelliasg.com/courses/grade-1-math/lessons/circle/

Audio narration is delivered exclusively via ElevenLabs (Voice: Alice,
Xb7hH8MSUJpSbSDYk0k2, eleven_multilingual_v2), following the full
pipeline architecture defined in audio_generation_pipeline.md. No
fallback to the browser Web Speech API is permitted.

5.2 Audio Map (src/utils/audioMap.js)

  Auto-generated. Maps exact text string → .mp3 relative path.
  Example entry:
    "Have you noticed? A coin is round. A clock is round. ..."
    → "/assets/audio/audio_wonder_have_you_noticed_..._0.mp3"

  Frontend checks audioMap[text] first. If found → static asset URL.
  If not found AND VITE_ELEVENLABS_API_KEY present → dynamic API call.
  If not found AND no key → narration silently skipped (no error shown).

5.3 Frontend Audio Engine (src/utils/audio.js)

  export async function getAudioUrl(text, style = 'statement') {
    // 1. Check static map
    if (audioMap[text]) {
      return audioMap[text];
    }

    // 2. Check in-memory cache
    if (elevenLabsCache[text]) {
      return elevenLabsCache[text];
    }

    // 3. Dynamic generation
    const apiKey = import.meta.env.VITE_ELEVENLABS_API_KEY;
    if (!apiKey) return null; // Silent skip — no browser fallback

    const response = await fetch(
      `https://api.elevenlabs.io/v1/text-to-speech/${VOICE_ID}`,
      {
        method: 'POST',
        headers: {
          'xi-api-key': apiKey,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          text,
          model_id: MODEL_ID,
          voice_settings: STYLE_SETTINGS[style],
        }),
      }
    );

    const blob = await response.blob();
    const url = URL.createObjectURL(blob);
    elevenLabsCache[text] = url;
    return url;
  }

  export async function speak(text, style = 'statement') {
    const url = await getAudioUrl(text, style);
    if (!url) return;
    const audio = new Audio(url);
    return audio.play();
  }

  // Narration segment array playback with eager preloading
  export async function narrate(segments, audioEnabled) {
    if (!audioEnabled) return;
    for (let i = 0; i < segments.length; i++) {
      // Preload next segment while current plays
      if (i + 1 < segments.length) {
        getAudioUrl(segments[i + 1].text, segments[i + 1].style);
      }
      await speak(segments[i].text, segments[i].style);
    }
  }

  Accuracy measurement (endDraw callback):
    const GUIDE_CENTRE = { x: 160, y: 160 };
    const GUIDE_RADIUS = 100;

    function measureAccuracy(points) {
      const deviations = points.map(p => {
        const dist = Math.hypot(p.x - GUIDE_CENTRE.x, p.y - GUIDE_CENTRE.y);
        return Math.abs(dist - GUIDE_RADIUS);
      });
      const avg = deviations.reduce((a, b) => a + b, 0) / deviations.length;
      return Math.max(0, 100 - (avg / GUIDE_RADIUS) * 100);
    }

  Completion criteria: accuracy ≥ 60% OR attempts ≥ 3
  On pass: dispatch COMPLETE_SIM_STATION(0), narrate celebration
  On fail: reset trail, narrate encouragement, increment attempts

  Touch support: touchstart/touchmove/touchend mirror mouse events
  (use event.touches[0].clientX/Y adjusted for canvas offset)

9. RANDOMISATION ENGINE

9.1 Fisher-Yates Shuffle (utils/shuffle.js)

  export function shuffleArray(array) {
    const arr = [...array];
    for (let i = arr.length - 1; i > 0; i--) {
      const j = Math.floor(Math.random() * (i + 1));
      [arr[i], arr[j]] = [arr[j], arr[i]];
    }
    return arr;
  }

  export function generateSessionQuestions(bank) {
    const byType = {};
    bank.forEach(q => {
      if (!byType[q.type]) byType[q.type] = [];
      byType[q.type].push(q);
    });
    // Pick exactly 10 from each type (shuffled)
    const selected = Object.values(byType)
      .flatMap(qs => shuffleArray(qs).slice(0, 10));
    return shuffleArray(selected);  // Final shuffle across all 100
  }

9.2 MCQ Option Shuffling

  // Options are shuffled on load so correct answer is never
  // always in position 0
  export function shuffleOptions(options, correctAnswer) {
    return shuffleArray(options); // correctAnswer is included in options array
  }

9.3 Pattern Question Generator (dynamic variant)

  export function generatePattern(difficulty) {
    const shapes = difficulty === 1
      ? ['circle', 'square']
      : difficulty === 2
        ? ['circle', 'square', 'triangle']
        : ['circle', 'square', 'triangle', 'rectangle'];

    const unit = shuffleArray(shapes).slice(0, difficulty === 1 ? 2 : 3);
    const length = difficulty === 1 ? 4 : difficulty === 2 ? 5 : 6;

    const sequence = [];
    for (let i = 0; i < length; i++) {
      sequence.push(unit[i % unit.length]);
    }
    sequence.push('?');
    return { sequence, answer: unit[length % unit.length] };
  }

9.4 Session Persistence

  const SESSION_KEY = 'intellia_circle_v1';

  // Restore on load (within 24 hours)
  const saved = JSON.parse(localStorage.getItem(SESSION_KEY));
  if (saved && Date.now() - saved.timestamp < 86400000) {
    dispatch({ type: ACTIONS.RESTORE_SESSION, payload: saved });
  }

  // Save on every state change (debounced 500ms)
  useEffect(() => {
    const timer = setTimeout(() => {
      localStorage.setItem(SESSION_KEY, JSON.stringify({
        phase: state.phase,
        currentQuestion: state.currentQuestion,
        xp: state.xp,
        badges: state.badges,
        worldScores: state.worldScores,
        simStationsComplete: state.simStationsComplete,
        timestamp: Date.now(),
      }));
    }, 500);
    return () => clearTimeout(timer);
  }, [state]);

10.1 XP Calculation (utils/scoring.js)

  export function calcXP(attemptNumber, hintsUsed, streak) {
    const base = attemptNumber === 1 ? 10 : hintsUsed > 0 ? 5 : 7;
    const streakBonus = streak >= 5 ? 5 : 0;
    return base + streakBonus;
  }

  export function calcStars(correct, total = 10) {
